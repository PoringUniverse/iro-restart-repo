
automacro movetoAmistrQuest {
	exclusive 1
	BaseLevel > 80
	run-once 1
	call {
	
		$questStatus = &config(Quest)
		
		switch ($questStatus) {
			case (== AmistrHuntMino) {
				
				do move morocc
				do conf lockMap moc_pryd05
			}
			case (== AmistrHuntGoat) {
				
				do move yuno
				do conf lockMap ein_fild06
			}
			case (== AmistrHuntMino) {
				
				do move morocc
				do conf lockMap moc_pryd05
			} else {
			
				do conf lockMap None
				do move prontera 144 &rand(190,192)
				do conf -f Quest Amistr
			
			}
		}
		
		
	}
}
