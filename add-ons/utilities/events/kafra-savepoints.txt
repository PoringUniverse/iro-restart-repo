# NOTICE : Set 0 or 1 to disable/enable autostorage or autosell

automacro CheckIfSaved {
	InMap prt_fild05, prontera, izlude, geffen, aldebaran, payon, morocc, alberta, comodo, cmd_fild07, yuno, louyang
	InSaveMap 0
	exclusive 1
	run-once 1
	macro_delay 2
	call ChangeSavePoints
}

automacro KafraSavePoints {
	InMap prt_fild05, prontera, izlude, geffen, aldebaran, payon, morocc, alberta, comodo, cmd_fild07, yuno, louyang
	exclusive 1
	run-once 1
	macro_delay 2
	call ChangeSavePoints 
}
